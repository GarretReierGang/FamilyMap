CREATE TABLE `SECURITY` 
( 
	`USER_ID` TEXT NOT NULL UNIQUE, 
	`SALT` TEXT DEFAULT 'AxrEsQesD' UNIQUE 
)

CREATE TABLE "USERS" 
( 
	`USER_NAME` TEXT NOT NULL UNIQUE, 
	`USER_ID` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, 
	`ME` INTEGER, 
	`EMAIL` TEXT NOT NULL UNIQUE, 
	`PASSWORD` TEXT NOT NULL 
)

CREATE TABLE "AUTH_TOKENS" 
( 
	`AUTH_ID` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, 
	`USER_ID` INTEGER NOT NULL, 
	FOREIGN KEY(`USER_ID`) REFERENCES `USERS`(`USER_ID`) 
)
CREATE TABLE "PERSON" 
( 
	`PERSON_ID` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, 
	`ASSOCIATED_USER` INTEGER NOT NULL,
	`FIRST_NAME` TEXT, 
	`LAST_NAME` TEXT, 
	`FATHER` INTEGER, 
	`MOTHER` INTEGER, 
	`SPOUSE` INTEGER, 
	FOREIGN KEY(`ASSOCIATED_USER`) REFERENCES `USERS`(`USER_ID`),
	FOREIGN KEY(`FATHER`) REFERENCES `PERSON`(`PERSON_ID`), 
	FOREIGN KEY(`MOTHER`) REFERENCES `PERSON`(`PERSON_ID`), 
	FOREIGN KEY(`SPOUSE`) REFERENCES `PERSON`(`PERSON_ID`) 
)
CREATE TABLE "EVENTS" 
( 
	`EVENT_ID` INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT UNIQUE, 
	`ASSOCIATED_USER` INTEGER,
	`LATITUDE` REAL NOT NULL, 
	`LONGITUDE` REAL NOT NULL, 
	`COUNTRY` INTEGER, `CITY` TEXT, 
	`EVENT_TYPE` TEXT, 
	`YEAR` TEXT NOT NULL, 
	`PERSON_ID` INTEGER, 
	FOREIGN KEY(`PERSON_ID`) REFERENCES `PERSON`(`PERSON_ID`), 
	FOREIGN KEY(`ASSOCIATED_USER`) REFERENCES `USERS`(`USER_ID`)
)
